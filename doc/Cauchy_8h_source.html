17 //
18 // Class to compute the generalized Cauchy point (GCP) for the L-BFGS-B algorithm,
19 // mainly for internal use.
20 //
21 // The target of the GCP procedure is to find a step size t such that
22 // x(t) = x0 - t * g is a local minimum of the quadratic function m(x),
23 // where m(x) is a local approximation to the objective function.
24 //
25 // First determine a sequence of break points t0=0, t1, t2, ..., tn.
26 // On each interval [t[i-1], t[i]], x is changing linearly.
27 // After passing a break point, one or more coordinates of x will be fixed at the bounds.
28 // We search the first local minimum of m(x) by examining the intervals [t[i-1], t[i]] sequentially.
29 //
30 // Reference:
31 // [1] R. H. Byrd, P. Lu, and J. Nocedal (1995). A limited memory algorithm for bound constrained optimization.
32 //
33 template <typename Scalar>
34 class ArgSort
35 {
36 private:
37  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;
38  typedef std::vector<int> IndexSet;
39 
40  const Scalar* values;
41 
42 public:
43  ArgSort(const Vector& value_vec) :
44  values(value_vec.data())
45  {}
46 
47  inline bool operator()(int key1, int key2) { return values[key1] < values[key2]; }
48  inline void sort_key(IndexSet& key_vec) const
49  {
50  std::sort(key_vec.begin(), key_vec.end(), *this);
51  }
52 };

54 template <typename Scalar>
55 class Cauchy
56 {
57 private:
58  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;
59  typedef Eigen::Matrix<int, Eigen::Dynamic, 1> IntVector;
60  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Matrix;
61  typedef std::vector<int> IndexSet;
62 
63  // Find the smallest index i such that brk[ord[i]] > t, assuming brk[ord] is already sorted.
64  // If the return value equals n, then all values are <= t.
65  static int search_greater(const Vector& brk, const IndexSet& ord, const Scalar& t, int start = 0)
66  {
67  const int nord = ord.size();
68  int i;
69  for(i = start; i < nord; i++)
70  {
71  if(brk[ord[i]] > t)
72  break;
73  }
74 
75  return i;
76  }
77 
78 public:
79  // bfgs: An object that represents the BFGS approximation matrix.
80  // x0: Current parameter vector.
81  // g: Gradient at x0.
82  // lb: Lower bounds for x.
83  // ub: Upper bounds for x.
84  // xcp: The output generalized Cauchy point.
85  // vecc: c = W'(xcp - x0), used in the subspace minimization routine.
86  // newact_set: Coordinates that newly become active during the GCP procedure.
87  // fv_set: Free variable set.
88  static void get_cauchy_point(
89  const BFGSMat<Scalar, true>& bfgs, const Vector& x0, const Vector& g, const Vector& lb, const Vector& ub,
90  Vector& xcp, Vector& vecc, IndexSet& newact_set, IndexSet& fv_set
91  )
92  {
93  // std::cout << "========================= Entering GCP search =========================\n\n";
94 
95  // Initialization
96  const int n = x0.size();
97  xcp.resize(n);
98  xcp.noalias() = x0;
99  vecc.resize(2 * bfgs.num_corrections());
100  vecc.setZero();
101  newact_set.clear();
102  newact_set.reserve(n);
103  fv_set.clear();
104  fv_set.reserve(n);
105 
106  // Construct break points
107  Vector brk(n), vecd(n);
108  // If brk[i] == 0, i belongs to active set
109  // If brk[i] == Inf, i belongs to free variable set
110  // Others are currently undecided
111  IndexSet ord;
112  ord.reserve(n);
113  const Scalar inf = std::numeric_limits<Scalar>::infinity();
114  for(int i = 0; i < n; i++)
115  {
116  if(g[i] < Scalar(0))
117  brk[i] = (x0[i] - ub[i]) / g[i];
118  else if(g[i] > Scalar(0))
119  brk[i] = (x0[i] - lb[i]) / g[i];
120  else
121  brk[i] = inf;
122 
123  const bool iszero = (brk[i] == Scalar(0));
124  vecd[i] = iszero ? Scalar(0) : -g[i];
125 
126  if(brk[i] == inf)
127  fv_set.push_back(i);
128  else if(!iszero)
129  ord.push_back(i);
130  }
131 
132  // Sort indices of break points
133  ArgSort<Scalar> sorting(brk);
134  sorting.sort_key(ord);
135 
136  // Break points `brko := brk[ord]` are in increasing order
137  // `ord` contains the coordinates that define the corresponding break points
138  // brk[i] == 0 <=> The i-th coordinate is on the boundary
139  const int nord = ord.size();
140  const int nfree = fv_set.size();
141  if( (nfree < 1) && (nord < 1) )
142  {
143  /* std::cout << "** All coordinates at boundary **\n";
144  std::cout << "\n========================= Leaving GCP search =========================\n\n"; */
145  return;
146  }
147 
148  // First interval: [il=0, iu=brk[ord[0]]]
149  // In case ord is empty, we take iu=Inf
150 
151  // p = W'd, c = 0
152  Vector vecp;
153  bfgs.apply_Wtv(vecd, vecp);
154  // f' = -d'd
155  Scalar fp = -vecd.squaredNorm();
156  // f'' = -theta * f' - p'Mp
157  Vector cache;
158  bfgs.apply_Mv(vecp, cache); // cache = Mp
159  Scalar fpp = -bfgs.theta() * fp - vecp.dot(cache);
160 
161  // Theoretical step size to move
162  Scalar deltatmin = -fp / fpp;
163 
164  // Limit on the current interval
165  Scalar il = Scalar(0);
166  // We have excluded the case that max(brk) <= 0
167  int b = 0;
168  Scalar iu = (nord < 1) ? inf : brk[ord[b]];
169  Scalar deltat = iu - il;
170 
171  /* int iter = 0;
172  std::cout << "** Iter " << iter << " **\n";
173  std::cout << " fp = " << fp << ", fpp = " << fpp << ", deltatmin = " << deltatmin << std::endl;
174  std::cout << " il = " << il << ", iu = " << iu << ", deltat = " << deltat << std::endl; */
175 
176  // If deltatmin >= deltat, we need to do the following things:
177  // 1. Update vecc
178  // 2. Since we are going to cross iu, the coordinates that define iu become active
179  // 3. Update some quantities on these new active coordinates (xcp, vecd, vecp)
180  // 4. Move to the next interval and compute the new deltatmin
181  bool crossed_all = false;
182  const int ncorr = bfgs.num_corrections();
183  Vector wact(2 * ncorr);
184  while(deltatmin >= deltat)
185  {
186  // Step 1
187  vecc.noalias() += deltat * vecp;
188 
189  // Step 2
190  // First check how many coordinates will be active when we cross the previous iu
191  // b is the smallest number such that brko[b] == iu
192  // Let bp be the largest number such that brko[bp] == iu
193  // Then coordinates ord[b] to ord[bp] will be active
194  const int act_begin = b;
195  const int act_end = search_greater(brk, ord, iu, b) - 1;
196 
197  // If nfree == 0 and act_end == nord-1, then we have crossed all coordinates
198  // We only need to update xcp from ord[b] to ord[bp], and then exit
199  if( (nfree == 0) && (act_end == nord - 1) )
200  {
201  // std::cout << "** [ ";
202  for(int i = act_begin; i <= act_end; i++)
203  {
204  const int act = ord[i];
205  xcp[act] = (vecd[act] > Scalar(0)) ? ub[act] : lb[act];
206  newact_set.push_back(act);
207  // std::cout << act + 1 << " ";
208  }
209  // std::cout << "] become active **\n\n";
210  // std::cout << "** All break points visited **\n\n";
211 
212  crossed_all = true;
213  break;
214  }
215 
216  // Step 3
217  // Update xcp and d on active coordinates
218  // std::cout << "** [ ";
219  fp += deltat * fpp;
220  for(int i = act_begin; i <= act_end; i++)
221  {
222  const int act = ord[i];
223  xcp[act] = (vecd[act] > Scalar(0)) ? ub[act] : lb[act];
224  // z = xcp - x0
225  const Scalar zact = xcp[act] - x0[act];
226  const Scalar gact = g[act];
227  const Scalar ggact = gact * gact;
228  wact.noalias() = bfgs.Wb(act);
229  bfgs.apply_Mv(wact, cache); // cache = Mw
230  fp += ggact + bfgs.theta() * gact * zact - gact * cache.dot(vecc);
231  fpp -= (bfgs.theta() * ggact + 2 * gact * cache.dot(vecp) + ggact * cache.dot(wact));
232  vecp.noalias() += gact * wact;
233  vecd[act] = Scalar(0);
234  newact_set.push_back(act);
235  // std::cout << act + 1 << " ";
236  }
237  // std::cout << "] become active **\n\n";
238 
239  // Step 4
240  // Theoretical step size to move
241  deltatmin = -fp / fpp;
242  // Update interval bound
243  il = iu;
244  b = act_end + 1;
245  // If we have visited all finite-valued break points, and have not exited earlier,
246  // then the next iu will be infinity. Simply exit the loop now
247  if(b >= nord)
248  break;
249  iu = brk[ord[b]];
250  // Width of the current interval
251  deltat = iu - il;
252 
253  /* iter++;
254  std::cout << "** Iter " << iter << " **\n";
255  std::cout << " fp = " << fp << ", fpp = " << fpp << ", deltatmin = " << deltatmin << std::endl;
256  std::cout << " il = " << il << ", iu = " << iu << ", deltat = " << deltat << std::endl; */
257  }
258 
259  // Last step
260  if(!crossed_all)
261  {
262  deltatmin = std::max(deltatmin, Scalar(0));
263  vecc.noalias() += deltatmin * vecp;
264  const Scalar tfinal = il + deltatmin;
265  // Update xcp on free variable coordinates
266  for(int i = 0; i < nfree; i++)
267  {
268  const int coord = fv_set[i];
269  xcp[coord] = x0[coord] + tfinal * vecd[coord];
270  }
271  for(int i = b; i < nord; i++)
272  {
273  const int coord = ord[i];
274  xcp[coord] = x0[coord] + tfinal * vecd[coord];
275  fv_set.push_back(coord);
276  }
277  }
278  // std::cout << "\n========================= Leaving GCP search =========================\n\n";
279  }
280 };
