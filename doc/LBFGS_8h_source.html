78  template <typename Foo>
79  inline int minimize(Foo& f, Vector& x, Scalar& fx)
80  {
81  using std::abs;
82 
83  // Dimension of the vector
84  const int n = x.size();
85  reset(n);
86 
87  // The length of lag for objective function value to test convergence
88  const int fpast = m_param.past;
89 
90  // Evaluate function and compute gradient
91  fx = f(x, m_grad);
92  Scalar xnorm = x.norm();
93  Scalar gnorm = m_grad.norm();
94  if(fpast > 0)
95  m_fx[0] = fx;
96 
97  // Early exit if the initial x is already a minimizer
98  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1)))
99  {
100  return 1;
101  }
102 
103  // Initial direction
104  m_drt.noalias() = -m_grad;
105  // Initial step size
106  Scalar step = Scalar(1) / m_drt.norm();
107 
108  // Number of iterations used
109  int k = 1;
110  for( ; ; )
111  {
112  // Save the curent x and gradient
113  m_xp.noalias() = x;
114  m_gradp.noalias() = m_grad;
115 
116  // Line search to update x, fx and gradient
117  LineSearch<Scalar>::LineSearch(f, fx, x, m_grad, step, m_drt, m_xp, m_param);
118 
119  // New x norm and gradient norm
120  xnorm = x.norm();
121  gnorm = m_grad.norm();
122 
123  // Convergence test -- gradient
124  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1)))
125  {
126  return k;
127  }
128  // Convergence test -- objective function value
129  if(fpast > 0)
130  {
131  const Scalar fxd = m_fx[k % fpast];
132  if(k >= fpast && abs(fxd - fx) <= m_param.delta * std::max(std::max(abs(fx), abs(fxd)), Scalar(1)))
133  return k;
134 
135  m_fx[k % fpast] = fx;
136  }
137  // Maximum number of iterations
138  if(m_param.max_iterations != 0 && k >= m_param.max_iterations)
139  {
140  return k;
141  }
142 
143  // Update s and y
144  // s_{k+1} = x_{k+1} - x_k
145  // y_{k+1} = g_{k+1} - g_k
146  m_bfgs.add_correction(x - m_xp, m_grad - m_gradp);
147 
148  // Recursive formula to compute d = -H * g
149  m_bfgs.apply_Hv(m_grad, -Scalar(1), m_drt);
150 
151  // Reset step = 1.0 as initial guess for the next line search
152  step = Scalar(1);
153  k++;
154  }
155 
156  return k;
157  }
