21  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;

23 public:
42  template <typename Foo>
43  static void LineSearch(Foo& f, Scalar& fx, Vector& x, Vector& grad,
44  Scalar& step,
45  const Vector& drt, const Vector& xp,
46  const LBFGSParam<Scalar>& param)
47  {
48  // Check the value of step
49  if(step <= Scalar(0))
50  std::invalid_argument("'step' must be positive");
51 
52  // Save the function value at the current x
53  const Scalar fx_init = fx;
54  // Projection of gradient on the search direction
55  const Scalar dg_init = grad.dot(drt);
56  // Make sure d points to a descent direction
57  if(dg_init > 0)
58  std::logic_error("the moving direction increases the objective function value");
59 
60  const Scalar test_decr = param.ftol * dg_init;
61 
62  // Upper and lower end of the current line search range
63  Scalar step_lo = 0,
64  step_hi = std::numeric_limits<Scalar>::infinity();
65 
66  for( int iter = 0; iter < param.max_linesearch; iter++ )
67  {
68  // x_{k+1} = x_k + step * d_k
69  x.noalias() = xp + step * drt;
70  // Evaluate this candidate
71  fx = f(x, grad);
72 
73  if(fx > fx_init + step * test_decr)
74  {
75  step_hi = step;
76  } else {
77  // Armijo condition is met
78  if(param.linesearch == LBFGS_LINESEARCH_BACKTRACKING_ARMIJO)
79  break;
80 
81  const Scalar dg = grad.dot(drt);
82  if(dg < param.wolfe * dg_init)
83  {
84  step_lo = step;
85  } else {
86  // Regular Wolfe condition is met
87  if(param.linesearch == LBFGS_LINESEARCH_BACKTRACKING_WOLFE)
88  break;
89 
90  if(dg > -param.wolfe * dg_init)
91  {
92  step_hi = step;
93  } else {
94  // Strong Wolfe condition is met
95  break;
96  }
97  }
98  }
99 
100  assert( step_lo < step_hi );
101 
102  if(iter >= param.max_linesearch)
103  throw std::runtime_error("the line search routine reached the maximum number of iterations");
104 
105  if(step < param.min_step)
106  throw std::runtime_error("the line search step became smaller than the minimum value allowed");
107 
108  if(step > param.max_step)
109  throw std::runtime_error("the line search step became larger than the maximum value allowed");
110 
111  // continue search in mid of current search range
112  step = std::isinf(step_hi) ? 2*step : step_lo/2 + step_hi/2;
113  }
114  }
115 };
