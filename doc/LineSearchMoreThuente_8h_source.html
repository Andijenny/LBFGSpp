33  // Mininum of a quadratic function that interpolates fa, ga, and fb
34  static Scalar quadratic_interp(const Scalar& a, const Scalar& b, const Scalar& fa, const Scalar& ga, const Scalar& fb)
35  {
36  const Scalar ba = b - a;
37  return a + Scalar(0.5) * ba * ba * ga / (fa - fb + ba * ga);
38  }
39 
40  // Mininum of a quadratic function that interpolates ga and gb
41  // Assume that ga != gb
42  static Scalar quadratic_interp(const Scalar& a, const Scalar& b, const Scalar& ga, const Scalar& gb)
43  {
44  return b + (b - a) * gb / (ga - gb);
45  }
46 
47  // Mininum of a cubic function that interpolates fa, ga, fb and gb
48  // Assume that a != b
49  static Scalar cubic_interp(const Scalar& a, const Scalar& b, const Scalar& fa, const Scalar& fb, const Scalar& ga, const Scalar& gb)
50  {
51  const Scalar ba = b - a;
52  const Scalar ba2 = ba * ba;
53  const Scalar ba3 = ba2 * ba;
54  const Scalar fba = fb - fa;
55  // q(x) = c0 + c1 * x + c2 * x^2 + c3 * x^3
56  // Solve c0, c1, c2, c3 such that q(a) = fa, q(b) = fb, q'(a) = ga, and q'(b) = gb
57  const Scalar c3 = (ga + gb) / ba2 - Scalar(2) * fba / ba3;
58  const Scalar c2 = fba / ba2 - ga / ba - (Scalar(2) * a + b) * c3;
59  const Scalar c1 = fba / ba - (a + b) * c2 - (a * a + a * b + b * b) * c3;
60  const Scalar c0 = fa - c1 * a - c2 * a * a - c3 * a * a * a;
61 
62  // q'(x) = c1 + 2 * c2 * x + 3 * c3 * x^2
63  // If q'(x) = 0 has no solution in [a, b], q(x) is monotone in [a, b]
64  // Case I: no solution globally
65  const Scalar delta = Scalar(4) * c2 * c2 - Scalar(12) * c1 * c3;
66  if(delta < Scalar(0))
67  return (fa < fb) ? a : b;
68  // Case II: no solution in (a, b)
69  const Scalar sdelta = std::sqrt(delta);
70  const Scalar sol1 = (-Scalar(2) * c2 - sdelta) / (Scalar(6) * c3);
71  const Scalar sol2 = (-Scalar(2) * c2 + sdelta) / (Scalar(6) * c3);
72  if( (sol1 - a) * (sol1 - b) >= Scalar(0) && (sol2 - a) * (sol2 - b) >= Scalar(0) )
73  return (fa < fb) ? a : b;
74 
75  // Now at least one solution is in (a, b)
76  // Check the second derivative
77  // q''(x) = 2 * c2 + 6 * c3 * x;
78  const Scalar qpp1 = Scalar(2) * c2 + Scalar(6) * c3 * sol1;
79  const Scalar sol = (qpp1 > Scalar(0)) ? sol1 : sol2;
80  // If the local minimum is not in [a, b], return one of the end points
81  if((sol - a) * (sol - b) >= Scalar(0))
82  return (fa < fb) ? a : b;
83 
84  // Compare the local minimum with the end points
85  const Scalar fsol = c0 + c1 * sol + c2 * sol * sol + c3 * sol * sol * sol;
86  Scalar res = (fa < fb) ? a : b;
87  res = (fsol < std::min(fa, fb)) ? sol : res;
88  return res;
89  }

91  static Scalar step_selection(
92  const Scalar& al, const Scalar& au, const Scalar& at,
93  const Scalar& fl, const Scalar& fu, const Scalar& ft,
94  const Scalar& gl, const Scalar& gu, const Scalar& gt
95  )
96  {
97  if(al == au)
98  return al;
99 
100  // ac: cubic interpolation of fl, ft, gl, gt
101  // aq: quadratic interpolation of fl, gl, ft
102  const Scalar ac = cubic_interp(al, at, fl, ft, gl, gt);
103  const Scalar aq = quadratic_interp(al, at, fl, gl, ft);
104  // Case 1: ft > fl
105  if(ft > fl)
106  return (std::abs(ac - al) < std::abs(aq - al)) ?
107  ac :
108  ((aq + ac) / Scalar(2));
109 
110  // as: quadratic interpolation of gl and gt
111  const Scalar as = quadratic_interp(al, at, gl, gt);
112  // Case 2: ft <= fl, gt * gl < 0
113  if(gt * gl < Scalar(0))
114  return (std::abs(ac - at) >= std::abs(as - at)) ? ac : as;
115 
116  // Case 3: ft <= fl, gt * gl >= 0, |gt| < |gl|
117  const Scalar delta = Scalar(0.66);
118  if(std::abs(gt) < std::abs(gl))
119  {
120  const Scalar res = (std::abs(ac - at) < std::abs(as - at)) ? ac : as;
121  return (at > al) ?
122  std::min(at + delta * (au - at), res) :
123  std::max(at + delta * (au - at), res);
124  }
125 
126  // ae: cubic interpolation of ft, fu, gt, gu
127  const Scalar ae = cubic_interp(at, au, ft, fu, gt, gu);
128  // Case 4: ft <= fl, gt * gl >= 0, |gt| >= |gl|
129  return (at > al) ?
130  std::min(at + delta * (au - at), ae) :
131  std::max(at + delta * (au - at), ae);
132  }

152  template <typename Foo>
153  static void LineSearch(Foo& f, Scalar& fx, Vector& x, Vector& grad,
154  Scalar& step, const Scalar& step_max,
155  const Vector& drt, const Vector& xp,
156  const LBFGSBParam<Scalar>& param)
157  {
158  // std::cout << "========================= Entering line search =========================\n\n";
159 
160  // Check the value of step
161  if(step <= Scalar(0))
162  std::invalid_argument("'step' must be positive");
163  if(step > step_max)
164  std::invalid_argument("'step' exceeds 'step_max'");
165 
166  // Save the function value at the current x
167  const Scalar fx_init = fx;
168  // Projection of gradient on the search direction
169  const Scalar dg_init = grad.dot(drt);
170 
171  // std::cout << "fx_init = " << fx_init << ", dg_init = " << dg_init << std::endl << std::endl;
172 
173  // Make sure d points to a descent direction
174  if(dg_init >= 0)
175  std::logic_error("the moving direction does not decrease the objective function value");
176 
177  // Tolerance for convergence test
178  // Sufficient decrease
179  const Scalar test_decr = param.ftol * dg_init;
180  // Curvature
181  const Scalar test_curv = -param.wolfe * dg_init;
182 
183  // The bracketing interval
184  Scalar I_lo = Scalar(0), I_hi = std::numeric_limits<Scalar>::infinity();
185  Scalar fI_lo = Scalar(0), fI_hi = std::numeric_limits<Scalar>::infinity();
186  Scalar gI_lo = (Scalar(1) - param.ftol) * dg_init, gI_hi = std::numeric_limits<Scalar>::infinity();
187  // Function value and gradient at the current step size
188  x.noalias() = xp + step * drt;
189  fx = f(x, grad);
190  Scalar dg = grad.dot(drt);
191 
192  // std::cout << "max_step = " << step_max << ", step = " << step << ", fx = " << fx << ", dg = " << dg << std::endl;
193 
194  // Convergence test
195  if(fx <= fx_init + step * test_decr && std::abs(dg) <= test_curv)
196  {
197  // std::cout << "** Criteria met\n\n";
198  // std::cout << "========================= Leaving line search =========================\n\n";
199  return;
200  }
201  if(step >= step_max)
202  {
203  // std::cout << "** Maximum step size reached\n\n";
204  // std::cout << "========================= Leaving line search =========================\n\n";
205  return;
206  }
207 
208  // Extrapolation factor
209  const Scalar delta = Scalar(1.1);
210  int iter;
211  for(iter = 0; iter < param.max_linesearch; iter++)
212  {
213  // ft = psi(step) = f(xp + step * drt) - f(xp) - step * test_decr
214  // gt = psi'(step) = dg - mu * dg_init
215  // mu = param.ftol
216  const Scalar ft = fx - fx_init - step * test_decr;
217  const Scalar gt = dg - param.ftol * dg_init;
218 
219  // Update bracketing interval and step size
220  if(ft > fI_lo)
221  {
222  // Case 1: ft > fl
223  const Scalar new_step = step_selection( I_lo, I_hi, step,
224  fI_lo, fI_hi, ft,
225  gI_lo, gI_hi, gt);
226  I_hi = step;
227  fI_hi = ft;
228  gI_hi = gt;
229 
230  step = new_step;
231 
232  // std::cout << "Case 1: new step = " << step;
233 
234  } else if(gt * (fI_lo - step) > Scalar(0)) {
235  // Case 2: ft <= fl, gt * (al - at) > 0
236  const Scalar new_step = std::min(step_max, step + delta * (step - I_lo));
237 
238  I_lo = step;
239  fI_lo = ft;
240  gI_lo = gt;
241 
242  step = new_step;
243 
244  // std::cout << "Case 2: new step = " << step;
245 
246  } else {
247  // Case 3: ft <= fl, gt * (al - at) <= 0
248  const Scalar new_step = step_selection( I_lo, I_hi, step,
249  fI_lo, fI_hi, ft,
250  gI_lo, gI_hi, gt);
251  I_hi = I_lo;
252  fI_hi = fI_lo;
253  gI_hi = gI_lo;
254 
255  I_lo = step;
256  fI_lo = ft;
257  gI_lo = gt;
258 
259  step = new_step;
260 
261  // std::cout << "Case 3: new step = " << step;
262  }
263 
264  if(step < param.min_step)
265  throw std::runtime_error("the line search step became smaller than the minimum value allowed");
266 
267  if(step > param.max_step)
268  throw std::runtime_error("the line search step became larger than the maximum value allowed");
269 
270  // Update parameter, function value, and gradient
271  x.noalias() = xp + step * drt;
272  fx = f(x, grad);
273  dg = grad.dot(drt);
274 
275  // std::cout << ", fx = " << fx << std::endl;
276 
277  // Convergence test
278  if(fx <= fx_init + step * test_decr && std::abs(dg) <= test_curv)
279  {
280  // std::cout << "** Criteria met\n\n";
281  // std::cout << "========================= Leaving line search =========================\n\n";
282  return;
283  }
284  if(step >= step_max)
285  {
286  // std::cout << "** Maximum step size reached\n\n";
287  // std::cout << "========================= Leaving line search =========================\n\n";
288  return;
289  }
290  }
291 
292  if(iter >= param.max_linesearch)
293  throw std::runtime_error("the line search routine reached the maximum number of iterations");
294  }
295 };
