23 class LineSearchNocedalWright
24 {
25 private:
26  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;

28 public:
45  template <typename Foo>
46  static void LineSearch(Foo& f, Scalar& fx, Vector& x, Vector& grad,
47  Scalar& step,
48  const Vector& drt, const Vector& xp,
49  const LBFGSParam<Scalar>& param)
50  {
51  // Check the value of step
52  if(step <= Scalar(0))
53  throw std::invalid_argument("'step' must be positive");
54 
55  if(param.linesearch != LBFGS_LINESEARCH_BACKTRACKING_STRONG_WOLFE)
56  throw std::invalid_argument("'param.linesearch' must be 'LBFGS_LINESEARCH_BACKTRACKING_STRONG_WOLFE' for LineSearchNocedalWright");
57 
58  // To make this implementation more similar to the other line search
59  // methods in LBFGSpp, the symbol names from the literature
60  // ("Numerical Optimizations") have been changed.
61  //
62  // Literature | LBFGSpp
63  // -----------|--------
64  // alpha | step
65  // phi | fx
66  // phi' | dg
67 
68  // the rate, by which the
69  const Scalar expansion = Scalar(2);
70 
71  // Save the function value at the current x
72  const Scalar fx_init = fx;
73  // Projection of gradient on the search direction
74  const Scalar dg_init = grad.dot(drt);
75  // Make sure d points to a descent direction
76  if(dg_init > 0)
77  throw std::logic_error("the moving direction increases the objective function value");
78 
79  const Scalar test_decr = param.ftol * dg_init, // Sufficient decrease
80  test_curv = -param.wolfe * dg_init; // Curvature
81 
82  // Ends of the line search range (step_lo > step_hi is allowed)
83  Scalar step_hi, step_lo = 0,
84  fx_hi, fx_lo = fx_init,
85  dg_hi, dg_lo = dg_init;
86 
87  // STEP 1: Bracketing Phase
88  // Find a range guaranteed to contain a step satisfying strong Wolfe.
89  //
90  // See also:
91  // "Numerical Optimization", "Algorithm 3.5 (Line Search Algorithm)".
92  int iter = 0;
93  for(;;)
94  {
95  x.noalias() = xp + step * drt;
96  fx = f(x, grad);
97 
98  if(iter++ >= param.max_linesearch)
99  return;
100 
101  const Scalar dg = grad.dot(drt);
102 
103  if( fx - fx_init > step * test_decr || (0 < step_lo && fx >= fx_lo) )
104  {
105  step_hi = step;
106  fx_hi = fx;
107  dg_hi = dg;
108  break;
109  }
110 
111  if( std::abs(dg) <= test_curv )
112  return;
113 
114  step_hi = step_lo;
115  fx_hi = fx_lo;
116  dg_hi = dg_lo;
117  step_lo = step;
118  fx_lo = fx;
119  dg_lo = dg;
120 
121  if( dg >= 0 )
122  break;
123 
124  step *= expansion;
125  }
126 
127  // STEP 2: Zoom Phase
128  // Given a range (step_lo,step_hi) that is guaranteed to
129  // contain a valid strong Wolfe step value, this method
130  // finds such a value.
131  //
132  // See also:
133  // "Numerical Optimization", "Algorithm 3.6 (Zoom)".
134  for(;;)
135  {
136  // use {fx_lo, fx_hi, dg_lo} to make a quadric interpolation of
137  // the function said interpolation is used to estimate the minimum
138  //
139  // polynomial: p (x) = c0*(x - step)² + c1
140  // conditions: p (step_hi) = fx_hi
141  // p (step_lo) = fx_lo
142  // p'(step_lo) = dg_lo
143  step = (fx_hi-fx_lo)*step_lo - (step_hi*step_hi - step_lo*step_lo)*dg_lo/2;
144  step /= (fx_hi-fx_lo) - (step_hi - step_lo )*dg_lo;
145 
146  // if interpolation fails, bisection is used
147  if( step <= std::min(step_lo,step_hi) ||
148  step >= std::max(step_lo,step_hi) )
149  step = step_lo/2 + step_hi/2;
150 
151  x.noalias() = xp + step * drt;
152  fx = f(x, grad);
153 
154  if(iter++ >= param.max_linesearch)
155  return;
156 
157  const Scalar dg = grad.dot(drt);
158 
159  if( fx - fx_init > step * test_decr || fx >= fx_lo )
160  {
161  if( step == step_hi )
162  throw std::runtime_error("the line search routine failed, possibly due to insufficient numeric precision");
163 
164  step_hi = step;
165  fx_hi = fx;
166  dg_hi = dg;
167  }
168  else
169  {
170  if( std::abs(dg) <= test_curv )
171  return;
172 
173  if( dg * (step_hi - step_lo) >= 0 )
174  {
175  step_hi = step_lo;
176  fx_hi = fx_lo;
177  dg_hi = dg_lo;
178  }
179 
180  if( step == step_lo )
181  throw std::runtime_error("the line search routine failed, possibly due to insufficient numeric precision");
182 
183  step_lo = step;
184  fx_lo = fx;
185  dg_lo = dg;
186  }
187  }
188  }
189 };
