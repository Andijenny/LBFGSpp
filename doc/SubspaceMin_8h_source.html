18 //
19 // Subspace minimization procedure of the L-BFGS-B algorithm,
20 // mainly for internal use.
21 //
22 // The target of subspace minimization is to minimize the quadratic function m(x)
23 // over the free variables, subject to the bound condition.
24 // Free variables stand for coordinates that are not at the boundary in xcp,
25 // the generalized Cauchy point.
26 //
27 // In the classical implementation of L-BFGS-B [1], the minimization is done by first
28 // ignoring the box constraints, followed by a line search. Our implementation is
29 // an exact minimization subject to the bounds, based on the BOXCQP algorithm [2].
30 //
31 // Reference:
32 // [1] R. H. Byrd, P. Lu, and J. Nocedal (1995). A limited memory algorithm for bound constrained optimization.
33 // [2] C. Voglis and I. E. Lagaris (2004). BOXCQP: An algorithm for bound constrained convex quadratic problems.
34 //
35 template <typename Scalar>
36 class SubspaceMin
37 {
38 private:
39  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, 1> Vector;
40  typedef Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> Matrix;
41  typedef std::vector<int> IndexSet;
42 
43  // v[ind]
44  static Vector subvec(const Vector& v, const IndexSet& ind)
45  {
46  const int nsub = ind.size();
47  Vector res(nsub);
48  for(int i = 0; i < nsub; i++)
49  res[i] = v[ind[i]];
50  return res;
51  }
52 
53  // v[ind] = rhs
54  static void subvec_assign(Vector& v, const IndexSet& ind, const Vector& rhs)
55  {
56  const int nsub = ind.size();
57  for(int i = 0; i < nsub; i++)
58  v[ind[i]] = rhs[i];
59  }
60 
61  // Test convergence of P set
62  static bool P_converged(const IndexSet& yP_set, const Vector& vecy, const Vector& vecl, const Vector& vecu)
63  {
64  const int nP = yP_set.size();
65  for(int i = 0; i < nP; i++)
66  {
67  const int coord = yP_set[i];
68  if(vecy[coord] < vecl[coord] || vecy[coord] > vecu[coord])
69  return false;
70  }
71  return true;
72  }
73 
74  // Test convergence of L set
75  static bool L_converged(const IndexSet& yL_set, const Vector& lambda)
76  {
77  const int nL = yL_set.size();
78  for(int i = 0; i < nL; i++)
79  {
80  const int coord = yL_set[i];
81  if(lambda[coord] < Scalar(0))
82  return false;
83  }
84  return true;
85  }
86 
87  // Test convergence of L set
88  static bool U_converged(const IndexSet& yU_set, const Vector& mu)
89  {
90  const int nU = yU_set.size();
91  for(int i = 0; i < nU; i++)
92  {
93  const int coord = yU_set[i];
94  if(mu[coord] < Scalar(0))
95  return false;
96  }
97  return true;
98  }
99 
100 public:
101  // bfgs: An object that represents the BFGS approximation matrix.
102  // x0: Current parameter vector.
103  // xcp: Computed generalized Cauchy point.
104  // g: Gradient at x0.
105  // lb: Lower bounds for x.
106  // ub: Upper bounds for x.
107  // Wd: W'(xcp - x0)
108  // newact_set: Coordinates that newly become active during the GCP procedure.
109  // fv_set: Free variable set.
110  // maxit: Maximum number of iterations.
111  // drt: The output direction vector, drt = xsm - x0.
112  static void subspace_minimize(
113  const BFGSMat<Scalar, true>& bfgs, const Vector& x0, const Vector& xcp, const Vector& g,
114  const Vector& lb, const Vector& ub, const Vector& Wd, const IndexSet& newact_set, const IndexSet& fv_set, int maxit,
115  Vector& drt
116  )
117  {
118  // std::cout << "========================= Entering subspace minimization =========================\n\n";
119 
120  // d = xcp - x0
121  drt.noalias() = xcp - x0;
122  // Size of free variables
123  const int nfree = fv_set.size();
124  // If there is no free variable, simply return drt
125  if(nfree < 1)
126  {
127  // std::cout << "========================= (Early) leaving subspace minimization =========================\n\n";
128  return;
129  }
130 
131  // std::cout << "New active set = [ "; for(std::size_t i = 0; i < newact_set.size(); i++) std::cout << newact_set[i] << " "; std::cout << "]\n";
132  // std::cout << "Free variable set = [ "; for(std::size_t i = 0; i < fv_set.size(); i++) std::cout << fv_set[i] << " "; std::cout << "]\n\n";
133 
134  // Compute F'BAb = -F'WMW'AA'd
135  Vector vecc(nfree);
136  bfgs.compute_FtBAb(fv_set, newact_set, Wd, drt, vecc);
137  // Set the vector y=F'd containing free variables, vector c=F'BAb+F'g for linear term,
138  // and vectors l and u for the new bounds
139  Vector vecy(nfree), vecl(nfree), vecu(nfree);
140  for(int i = 0; i < nfree; i++)
141  {
142  const int coord = fv_set[i];
143  vecy[i] = drt[coord];
144  vecl[i] = lb[coord] - x0[coord];
145  vecu[i] = ub[coord] - x0[coord];
146  vecc[i] += g[coord];
147  }
148 
149  // Dual variables
150  Vector lambda = Vector::Zero(nfree), mu = Vector::Zero(nfree);
151 
152  // Iterations
153  IndexSet L_set, U_set, P_set, yL_set, yU_set, yP_set;
154  L_set.reserve(nfree / 3); yL_set.reserve(nfree / 3);
155  U_set.reserve(nfree / 3); yU_set.reserve(nfree / 3);
156  P_set.reserve(nfree); yP_set.reserve(nfree);
157  int k;
158  for(k = 0; k < maxit; k++)
159  {
160  // Construct the L, U, and P sets, and then update values
161  // Indices in original drt vector
162  L_set.clear();
163  U_set.clear();
164  P_set.clear();
165  // Indices in y
166  yL_set.clear();
167  yU_set.clear();
168  yP_set.clear();
169  for(int i = 0; i < nfree; i++)
170  {
171  const int coord = fv_set[i];
172  const Scalar li = vecl[i], ui = vecu[i];
173  if( (vecy[i] < li) || (vecy[i] == li && lambda[i] >= Scalar(0)) )
174  {
175  L_set.push_back(coord);
176  yL_set.push_back(i);
177  vecy[i] = li;
178  mu[i] = Scalar(0);
179  } else if( (vecy[i] > ui) || (vecy[i] == ui && mu[i] >= Scalar(0)) ) {
180  U_set.push_back(coord);
181  yU_set.push_back(i);
182  vecy[i] = ui;
183  lambda[i] = Scalar(0);
184  } else {
185  P_set.push_back(coord);
186  yP_set.push_back(i);
187  lambda[i] = Scalar(0);
188  mu[i] = Scalar(0);
189  }
190  }
191 
192  /* std::cout << "** Iter " << k << " **\n";
193  std::cout << " L = [ "; for(std::size_t i = 0; i < L_set.size(); i++) std::cout << L_set[i] << " "; std::cout << "]\n";
194  std::cout << " U = [ "; for(std::size_t i = 0; i < U_set.size(); i++) std::cout << U_set[i] << " "; std::cout << "]\n";
195  std::cout << " P = [ "; for(std::size_t i = 0; i < P_set.size(); i++) std::cout << P_set[i] << " "; std::cout << "]\n\n"; */
196 
197  // Split the W matrix according to P, L, and U
198  Matrix WP = bfgs.Wb(P_set);
199  // Solve y[P] = -inv(B[P, P]) * (B[P, L] * l[L] + B[P, U] * u[U] + c[P])
200  const int nP = P_set.size();
201  if(nP > 0)
202  {
203  Vector rhs = subvec(vecc, yP_set);
204  Vector lL = subvec(vecl, yL_set);
205  Vector uU = subvec(vecu, yU_set);
206  Vector tmp(nP);
207  bool nonzero = bfgs.apply_PtBQv(WP, L_set, lL, tmp, true);
208  if(nonzero)
209  rhs.noalias() += tmp;
210  nonzero = bfgs.apply_PtBQv(WP, U_set, uU, tmp, true);
211  if(nonzero)
212  rhs.noalias() += tmp;
213 
214  bfgs.solve_PtBP(WP, -rhs, tmp);
215  subvec_assign(vecy, yP_set, tmp);
216  }
217 
218  // Solve lambda[L] = B[L, F] * y + c[L]
219  const int nL = L_set.size();
220  const int nU = U_set.size();
221  Vector Fy;
222  if(nL > 0 || nU > 0)
223  bfgs.apply_WtPv(fv_set, vecy, Fy);
224  if(nL > 0)
225  {
226  Vector res;
227  bfgs.apply_PtWMv(L_set, Fy, res, Scalar(-1));
228  res.noalias() += subvec(vecc, yL_set);
229  subvec_assign(lambda, yL_set, res);
230  }
231 
232  // Solve mu[U] = -B[U, F] * y - c[U]
233  if(nU > 0)
234  {
235  Vector res;
236  bfgs.apply_PtWMv(U_set, Fy, res, Scalar(-1));
237  res.noalias() = -res - subvec(vecc, yU_set);
238  subvec_assign(mu, yU_set, res);
239  }
240 
241  // Test convergence
242  if( L_converged(yL_set, lambda) && U_converged(yU_set, mu) && P_converged(yP_set, vecy, vecl, vecu) )
243  break;
244  }
245 
246  if(k >= maxit)
247  throw std::runtime_error("the subspace minimization routine reached the maximum number of iterations");
248 
249  // std::cout << "** Minimization finished in " << k + 1 << " iteration(s) **\n\n";
250  // std::cout << "========================= Leaving subspace minimization =========================\n\n";
251 
252  subvec_assign(drt, fv_set, vecy);
253  }
254 };
